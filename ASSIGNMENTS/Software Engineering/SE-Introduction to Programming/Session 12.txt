Q-1 Declare a structure named Playlist to store details of a song: title (string), artist (string), and duration in seconds (integer). Initialize one Playlist variable with your favorite song's details and print each field.
Ans.
#include <stdio.h>

// Declare structure
struct Playlist {
    char title[50];
    char artist[50];
    int duration;
};

int main() {
    // Initialize structure variable
    struct Playlist song = {"O Maahi O Maahi", "Arijit Singh", 233};

    // Print structure fields
    printf("Song Title : %s\n", song.title);
    printf("Artist     : %s\n", song.artist);
    printf("Duration   : %d seconds\n", song.duration);

    return 0;
}

Output:
Song Title : O Maahi O Maahi
Artist     : Arijit Singh
Duration   : 233 seconds





Q-2 Create a structure called FoodItem to store Zomato-style menu data: itemName (string), price (float), and rating (float). Initialize an array of 3 FoodItem variables with real menu items and display their details using a loop.
Ans.
#include <stdio.h>

// Declare structure
struct FoodItem {
    char itemName[50];
    float price;
    float rating;
};

int main() {
    // Initialize array of structures
    struct FoodItem menu[3] = {
        {"Pizza", 299.00, 4.5},
        {"Burger", 149.00, 4.3},
        {"Frankie", 99.00, 4.6}
    };

    // Display menu details
    printf("Zomato Menu:\n\n");

    for (int i = 0; i < 3; i++) {
        printf("Food Item %d\n", i + 1);
        printf("Item Name : %s\n", menu[i].itemName);
        printf("Price     : Rs. %.2f\n", menu[i].price);
        printf("Rating    : %.1f\n\n", menu[i].rating);
    }

    return 0;
}

Output:
Zomato Menu:

Food Item 1
Item Name : Pizza
Price     : Rs. 299.00
Rating    : 4.5

Food Item 2
Item Name : Burger
Price     : Rs. 149.00
Rating    : 4.3

Food Item 3
Item Name : Frankie
Price     : Rs. 99.00
Rating    : 4.6





Q-3 Define a nested structure called MovieShow for a BookMyShow-style app: Movie (string), Screen (integer), and a nested structure Time with hours and minutes (integers). Create and initialize a MovieShow variable for any movie and print its details in the format 'Movie: X, Screen: Y, Time: HH:MM'.
Ans.#include <stdio.h>

// Nested structure for Time
struct Time {
    int hours;
    int minutes;
};

// Main structure
struct MovieShow {
    char movie[50];
    int screen;
    struct Time showTime;
};

int main() {
    // Initialize MovieShow structure
    struct MovieShow show = {"Alpha", 2, {8, 15}};

    // Print movie details
    printf("Movie: %s\n", show.movie);
    printf("Screen: %d\n", show.screen);
    printf("Time: %02d:%02d\n", show.showTime.hours, show.showTime.minutes);

    return 0;
}

Output:
Movie: Alpha
Screen: 2
Time: 08:15





Q-4 Build a structure called InstaProfile with fields: username (string), followers (integer), and a nested structure Bio with fields: description (string) and age (integer). Initialize an InstaProfile variable with your own details and display all fields.
Ans.
#include <stdio.h>

// Nested structure for Bio
struct Bio {
    char description[100];
    int age;
};

// Main structure
struct InstaProfile {
    char username[50];
    int followers;
    struct Bio bio;
};

int main() {
    // Initialize InstaProfile structure
    struct InstaProfile profile = {
        "Krishna Patel",
        200,
        {"Computer Engineering Student", 23}
    };

    // Display profile details
    printf("Username    : %s\n", profile.username);
    printf("Followers   : %d\n", profile.followers);
    printf("Description : %s\n", profile.bio.description);
    printf("Age         : %d\n", profile.bio.age);

    return 0;
}

Output:
Username    : Krishna Patel
Followers   : 200
Description : Computer Engineering Student
Age         : 23
