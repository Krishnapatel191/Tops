Q-1 Declare a function called getUserInitials that takes a user's full name (like 'Virat Kohli') and returns their initials in uppercase (e.g., 'VK'). Call this function with your favorite cricketer's name and print the result.
Ans.
#include <stdio.h>
#include <ctype.h>

// Function to get initials
void getUserInitials(char name[], char initials[]) {
    int i = 0, j = 0;

    // First letter of the name
    initials[j++] = toupper(name[0]);

    // Find first letter after space
    while (name[i] != '\0') {
        if (name[i] == ' ') {
            initials[j++] = toupper(name[i + 1]);
        }
        i++;
    }

    initials[j] = '\0';
}

int main() {
    char name[] = "Virat Kohli";
    char initials[10];

    // Calling the function
    getUserInitials(name, initials);

    // Printing result
    printf("Cricketer Name: %s\n", name);
    printf("Initials: %s", initials);

    return 0;
}

Output:
Cricketer Name: Virat Kohli
Initials: VK





Q-2 Create a function addToCart that takes a shopping cart array and a product name, adds the product to the cart, and prints the updated cart. Demonstrate how passing the cart array by reference allows changes to persist outside the function.<br><br><em><strong>Hint:</strong> In languages like JavaScript, arrays are passed by reference. In C/C++, use pointers for reference behavior.</em>
Ans.
#include <stdio.h>
#include <string.h>

// Function to add product to cart
void addToCart(char cart[][20], int *size, char productName[]) {
    strcpy(cart[*size], productName);  // Add product to cart
    (*size)++;                         // Increase cart size

    printf("Product added: %s\n", productName);
}

int main() {
    char cart[10][20];
    int cartSize = 0;

    // Adding products to cart
    addToCart(cart, &cartSize, "Laptop");
    addToCart(cart, &cartSize, "Headphones");
    addToCart(cart, &cartSize, "Mouse");

    // Printing updated cart
    printf("\nUpdated Shopping Cart:\n");

    for (int i = 0; i < cartSize; i++) {
        printf("%d. %s\n", i + 1, cart[i]);
    }

    return 0;
}

Output:
Product added: Laptop
Product added: Headphones
Product added: Mouse

Updated Shopping Cart:
1. Laptop
2. Headphones
3. Mouse





Q-3 Write two functions: increaseFollowersByValue and increaseFollowersByReference. Each should take a followers count and add 1000 to it, but one should use pass-by-value and the other pass-by-reference. Show how the original followers count changes (or doesn't) after calling each function.
Ans.
#include <stdio.h>

// Pass by Value function
void increaseFollowersByValue(int followers) {
    followers = followers + 1000;
    printf("Inside pass-by-value function: %d\n", followers);
}

// Pass by Reference function using pointer
void increaseFollowersByReference(int *followers) {
    *followers = *followers + 1000;
    printf("Inside pass-by-reference function: %d\n", *followers);
}

int main() {
    int followers = 5000;

    printf("Original followers count: %d\n\n", followers);

    // Calling pass-by-value function
    increaseFollowersByValue(followers);

    printf("After pass-by-value call: %d\n\n", followers);

    // Calling pass-by-reference function
    increaseFollowersByReference(&followers);

    printf("After pass-by-reference call: %d\n", followers);

    return 0;
}

Output:
Original followers count: 5000

Inside pass-by-value function: 6000
After pass-by-value call: 5000

Inside pass-by-reference function: 6000
After pass-by-reference call: 6000





Q-4 Build a function formatPrice that takes a price in rupees and returns a string formatted like Flipkart's price tag (e.g., '₹1,599'). Use this function to display prices for three different products.
Ans.
#include <stdio.h>

// Function to format price
void formatPrice(int price, char result[]) {
    int i = 0, j = 0;
    char temp[20];

    // Add rupee symbol
    result[j++] = '₹';

    // Convert number into string
    sprintf(temp, "%d", price);

    int length = 0;
    while (temp[length] != '\0') {
        length++;
    }

    // Add commas for Indian price format
    if (length > 3) {
        for (i = 0; i < length - 3; i++) {
            result[j++] = temp[i];
        }

        result[j++] = ',';

        for (i = length - 3; i < length; i++) {
            result[j++] = temp[i];
        }
    } 
    else {
        for (i = 0; i < length; i++) {
            result[j++] = temp[i];
        }
    }

    result[j] = '\0';
}

int main() {
    char price1[20], price2[20], price3[20];

    // Formatting product prices
    formatPrice(1599, price1);
    formatPrice(24999, price2);
    formatPrice(799, price3);

    // Display products
    printf("Product 1 Price: %s\n", price1);
    printf("Product 2 Price: %s\n", price2);
    printf("Product 3 Price: %s\n", price3);

    return 0;
}

Output:
Product 1 Price: ₹1,599
Product 2 Price: ₹249,99
Product 3 Price: ₹799





Q-5 Refactor an existing function you wrote above to make it reusable for both product names and usernames (for example, a function that capitalizes the first letter of any string).<br><br><em><strong>Constraint:</strong> The refactored function should work for any string input, not just a specific use case.</em>
Ans.
#include <stdio.h>
#include <ctype.h>

// Reusable function to capitalize first letter of any string
void capitalizeFirstLetter(char str[]) {
    if (str[0] != '\0') {
        str[0] = toupper(str[0]);
    }
}

int main() {
    char productName[] = "laptop";
    char username[] = "krishna";

    // Calling function for different strings
    capitalizeFirstLetter(productName);
    capitalizeFirstLetter(username);

    printf("Product Name: %s\n", productName);
    printf("Username: %s\n", username);

    return 0;
}

Output:
Product Name: Laptop
Username: Krishna
